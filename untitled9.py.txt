import numpy as np
import matplotlib.pyplot as plt
x = np.linspace(0, 4 * np.pi, 1000)
y = np.sin(x)

plt.figure(figsize=(10, 6))  # Grafikning o'lchamini belgilash
plt.plot(x, y, label='sin(x)', color='blue')


plt.title("Sinus funksiyasining grafigi")
plt.xlabel("x qiymatlari (rad)")
plt.ylabel("sin(x)")


plt.grid(True)
plt.legend()

plt.show()
